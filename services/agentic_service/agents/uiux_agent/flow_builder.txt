from agents.uiux_agent.schemas import UserFlow, FlowNode, FlowEdge, UIScreen


def _contains_any(text: str | None, keywords: list[str]) -> bool:
    """
    Small helper to detect requested UI/UX changes from a user prompt.
    """
    if not text:
        return False

    text = text.lower()
    return any(keyword.lower() in text for keyword in keywords)


def build_default_screens(
    srs: dict,
    api_contract: dict,
    include_admin: bool = True,
    change_request: str | None = None
) -> list[UIScreen]:
    """
    Builds the e-commerce screen inventory from approved SRS + API contract.

    The important upgrade:
    - If the user requests a UI change, this function can add/change screens.
    - The revised screen inventory is then used by BOTH:
      1) user flow generation
      2) wireframe generation

    This makes revision consistent.
    """

    frs = srs.get("functional_requirements", [])

    def match_requirements(*keywords: str) -> list[str]:
        matched = []

        for fr in frs:
            text = f"{fr.get('title', '')} {fr.get('description', '')}".lower()

            if any(keyword.lower() in text for keyword in keywords):
                matched.append(fr.get("id", ""))

        return matched or [fr.get("id", "") for fr in frs[:1]]

    screens = [
        UIScreen(
            id="UI-SCR-01",
            name="Login",
            file_name="UI-SCR-01_login.html",
            description="Login and account access screen.",
            route="/login",
            related_requirements=match_requirements("login", "register", "user", "account"),
        ),
        UIScreen(
            id="UI-SCR-02",
            name="Catalog",
            file_name="UI-SCR-02_catalog.html",
            description="Product catalog browsing screen with search and filters.",
            route="/products",
            related_requirements=match_requirements("catalog", "browse", "product", "search"),
        ),
        UIScreen(
            id="UI-SCR-03",
            name="Product Details",
            file_name="UI-SCR-03_product_details.html",
            description="Detailed product view with add-to-cart action.",
            route="/products/{product_id}",
            related_requirements=match_requirements("product details", "product", "add to cart"),
        ),
        UIScreen(
            id="UI-SCR-04",
            name="Cart",
            file_name="UI-SCR-04_cart.html",
            description="Shopping cart screen for reviewing and updating items.",
            route="/cart",
            related_requirements=match_requirements("cart"),
        ),
        UIScreen(
            id="UI-SCR-05",
            name="Checkout",
            file_name="UI-SCR-05_checkout.html",
            description="Checkout screen with address, payment method, and order summary.",
            route="/checkout",
            related_requirements=match_requirements("checkout", "payment"),
        ),
        UIScreen(
            id="UI-SCR-06",
            name="Order Confirmation",
            file_name="UI-SCR-06_order_confirmation.html",
            description="Order success confirmation screen.",
            route="/order-confirmation",
            related_requirements=match_requirements("order", "place"),
        ),
        UIScreen(
            id="UI-SCR-07",
            name="Order History",
            file_name="UI-SCR-07_order_history.html",
            description="Customer order history screen.",
            route="/orders",
            related_requirements=match_requirements("order history", "history", "orders"),
        ),
    ]

    if include_admin:
        screens.append(
            UIScreen(
                id="UI-SCR-08",
                name="Admin Product Management",
                file_name="UI-SCR-08_admin_products.html",
                description="Admin screen for managing product records.",
                route="/admin/products",
                related_requirements=match_requirements("admin", "manage product", "product management"),
            )
        )

    # ---------------------------------------------------------
    # Revision-driven screens
    # ---------------------------------------------------------
    # These screens are added when the user asks for a change.
    # This updates BOTH the flow and wireframes.
    # ---------------------------------------------------------

    if _contains_any(change_request, ["payment failed", "failed payment", "payment error", "checkout error", "error state"]):
        screens.append(
            UIScreen(
                id="UI-SCR-09",
                name="Checkout Error State",
                file_name="UI-SCR-09_checkout_error.html",
                description="Checkout error state screen shown when mock payment fails or validation fails.",
                route="/checkout/error",
                related_requirements=match_requirements("checkout", "payment", "error"),
            )
        )

    if _contains_any(change_request, ["wishlist", "save for later"]):
        screens.append(
            UIScreen(
                id="UI-SCR-10",
                name="Wishlist",
                file_name="UI-SCR-10_wishlist.html",
                description="Wishlist screen for saved products.",
                route="/wishlist",
                related_requirements=match_requirements("wishlist", "product", "customer"),
            )
        )

    if _contains_any(change_request, ["coupon", "discount", "promo"]):
        # Coupon does not need a separate screen. It updates checkout wireframe.
        # We keep the screen inventory unchanged but flow/wireframe will show coupon step.
        pass

    return screens


def build_user_flow(
    screens: list[UIScreen],
    change_request: str | None = None
) -> UserFlow:
    """
    Builds the main customer purchase flow.

    Upgrade:
    - If change_request includes checkout error/payment failure,
      the flow gets an error branch.
    - If change_request includes wishlist, the flow gets a wishlist branch.
    - If change_request includes coupon/discount, the flow gets a coupon validation node.

    This makes the flow match the revised wireframes.
    """

    screen_map = {screen.id: screen for screen in screens}

    def reqs(screen_id: str) -> list[str]:
        if screen_id in screen_map:
            return screen_map[screen_id].related_requirements
        return []

    nodes = [
        FlowNode(id="FLOW-N01", label="Start", screen_id=None),
        FlowNode(id="FLOW-N02", label="Login / Continue", screen_id="UI-SCR-01", related_requirements=reqs("UI-SCR-01")),
        FlowNode(id="FLOW-N03", label="Browse Catalog", screen_id="UI-SCR-02", related_requirements=reqs("UI-SCR-02")),
        FlowNode(id="FLOW-N04", label="View Product Details", screen_id="UI-SCR-03", related_requirements=reqs("UI-SCR-03")),
        FlowNode(id="FLOW-N05", label="Manage Cart", screen_id="UI-SCR-04", related_requirements=reqs("UI-SCR-04")),
        FlowNode(id="FLOW-N06", label="Checkout", screen_id="UI-SCR-05", related_requirements=reqs("UI-SCR-05")),
        FlowNode(id="FLOW-N07", label="Order Confirmation", screen_id="UI-SCR-06", related_requirements=reqs("UI-SCR-06")),
        FlowNode(id="FLOW-N08", label="View Order History", screen_id="UI-SCR-07", related_requirements=reqs("UI-SCR-07")),
    ]

    edges = [
        FlowEdge(from_node="FLOW-N01", to_node="FLOW-N02"),
        FlowEdge(from_node="FLOW-N02", to_node="FLOW-N03", condition="User authenticated or guest allowed"),
        FlowEdge(from_node="FLOW-N03", to_node="FLOW-N04", condition="Select product"),
        FlowEdge(from_node="FLOW-N04", to_node="FLOW-N05", condition="Add to cart"),
        FlowEdge(from_node="FLOW-N05", to_node="FLOW-N06", condition="Proceed to checkout"),
        FlowEdge(from_node="FLOW-N06", to_node="FLOW-N07", condition="Mock payment success"),
        FlowEdge(from_node="FLOW-N07", to_node="FLOW-N08", condition="View orders"),
    ]

    # Add checkout error branch
    if "UI-SCR-09" in screen_map:
        nodes.append(
            FlowNode(
                id="FLOW-N09",
                label="Checkout Error / Retry Payment",
                screen_id="UI-SCR-09",
                related_requirements=reqs("UI-SCR-09"),
            )
        )

        edges.append(
            FlowEdge(
                from_node="FLOW-N06",
                to_node="FLOW-N09",
                condition="Mock payment failed or validation error",
            )
        )
        edges.append(
            FlowEdge(
                from_node="FLOW-N09",
                to_node="FLOW-N06",
                condition="Retry checkout",
            )
        )

    # Add wishlist branch
    if "UI-SCR-10" in screen_map:
        nodes.append(
            FlowNode(
                id="FLOW-N10",
                label="Wishlist",
                screen_id="UI-SCR-10",
                related_requirements=reqs("UI-SCR-10"),
            )
        )

        edges.append(
            FlowEdge(
                from_node="FLOW-N04",
                to_node="FLOW-N10",
                condition="Save for later",
            )
        )
        edges.append(
            FlowEdge(
                from_node="FLOW-N10",
                to_node="FLOW-N04",
                condition="Open saved product",
            )
        )

    # Add coupon validation node to the flow
    if _contains_any(change_request, ["coupon", "discount", "promo"]):
        nodes.append(
            FlowNode(
                id="FLOW-N11",
                label="Apply Coupon / Validate Discount",
                screen_id="UI-SCR-05",
                related_requirements=reqs("UI-SCR-05"),
            )
        )

        edges.append(
            FlowEdge(
                from_node="FLOW-N05",
                to_node="FLOW-N11",
                condition="Apply coupon",
            )
        )
        edges.append(
            FlowEdge(
                from_node="FLOW-N11",
                to_node="FLOW-N06",
                condition="Coupon validated",
            )
        )

    return UserFlow(
        id="UF-001",
        name="Customer Purchase Flow",
        actor="Customer",
        nodes=nodes,
        edges=edges,
    )


def build_mermaid_from_flow(flow: UserFlow) -> str:
    """
    Converts structured UserFlow into Mermaid syntax.

    Mermaid is generated programmatically to avoid invalid LLM syntax.
    """

    lines = ["flowchart TD"]

    for node in flow.nodes:
        label = node.label

        if node.screen_id:
            label = f"{node.screen_id} {node.label}"

        safe_label = label.replace('"', "'")

        if node.id == "FLOW-N01":
            lines.append(f'    {node.id}(["{safe_label}"])')
        else:
            lines.append(f'    {node.id}["{safe_label}"]')

    lines.append("")

    for edge in flow.edges:
        if edge.condition:
            condition = edge.condition.replace('"', "'")
            lines.append(f'    {edge.from_node} -->|"{condition}"| {edge.to_node}')
        else:
            lines.append(f"    {edge.from_node} --> {edge.to_node}")

    return "\n".join(lines)
