from agents.uiux_agent.schemas import UIScreen


def _contains_any(text: str | None, keywords: list[str]) -> bool:
    """
    Checks whether the user's change request contains any keyword.
    """
    if not text:
        return False

    text = text.lower()
    return any(keyword.lower() in text for keyword in keywords)


def _revision_notice(change_request: str | None) -> str:
    """
    Adds a visible note when the screen was generated as part of a revision.
    """
    if not change_request:
        return ""

    safe_request = change_request.replace("<", "").replace(">", "")

    return f"""
    <div class="mb-6 rounded-xl border border-amber-300 bg-amber-50 p-4 text-sm text-amber-900">
        <strong>Revision applied:</strong> {safe_request}
    </div>
    """

def fallback_wireframe_html(
    project_name: str,
    screen: UIScreen,
    change_request: str | None = None
) -> str:
    """
    Improved low-fidelity wireframes with more realistic structure and visible details.
    """

    related = ", ".join(screen.related_requirements)

    checkout_error_requested = _contains_any(
        change_request,
        ["payment failed", "failed payment", "payment error", "checkout error", "error state"],
    )

    coupon_requested = _contains_any(
        change_request,
        ["coupon", "discount", "promo"],
    )

    wishlist_requested = _contains_any(
