const mongoose = require('mongoose');
const Schema = mongoose.Schema;

// --- 1. User Schema ---
const UserSchema = new Schema({
    name: { type: String, required: true },
    email: { type: String, required: true, unique: true },
    password_hash: { type: String, required: true },
    role: { type: String, enum: ['customer', 'admin'], default: 'customer' },
    created_at: { type: Date, default: Date.now }
});
const User = mongoose.model('User', UserSchema);

// --- 2. Category Schema ---
const CategorySchema = new Schema({
    name: { type: String, required: true, unique: true },
    description: { type: String }
});
const Category = mongoose.model('Category', CategorySchema);

// --- 3. Product Schema ---
const ProductSchema = new Schema({
    category_id: { type: mongoose.Schema.Types.ObjectId, ref: 'Category', required: true },
    name: { type: String, required: true },
    description: { type: String, required: true },
    price: { type: Number, required: true }, // Using Number for decimal representation
    stock_quantity: { type: Number, required: true, default: 0 },
    image_url: { type: String },
    is_active: { type: Boolean, default: true }
});
const Product = mongoose.model('Product', ProductSchema);

// --- 4. Address Schema ---
const AddressSchema = new Schema({
    line1: { type: String, required: true },
    city: { type: String, required: true },
    postal_code: { type: String, required: true },
    country: { type: String, required: true }
});
// Embedding Address within User for simplicity in MVP
UserSchema.virtual('address').embedSchema(AddressSchema);


// --- 5. Cart Schema ---
const CartSchema = new Schema({
    user_id: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true, unique: true },
    created_at: { type: Date, default: Date.now },
    updated_at: { type: Date, default: Date.now }
}, { timestamps: true });
const Cart = mongoose.model('Cart', CartSchema);

// --- 6. CartItem Schema ---
const CartItemSchema = new Schema({
    product_id: { type: mongoose.Schema.Types.ObjectId, ref: 'Product', required: true },
    quantity: { type: Number, required: true, min: 1 },
    unit_price: { type: Number, required: true }
}, { _id: false }); // Composite key handling
CartSchema.virtual('items').add({ type: [CartItemSchema] });
const CartItem = mongoose.model('CartItem', CartSchema); // Using Cart model for items

// --- 7. Order Schema ---
const OrderSchema = new Schema({
    user_id: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
    status: { type: String, enum: ['pending', 'processing', 'shipped', 'delivered', 'cancelled'], default: 'pending' },
    total_amount: { type: Number, required: true },
    shipping_address: { type: Schema.Types.Mixed, required: true }, // Using mixed type for embedded address
    created_at: { type: Date, default: Date.now }
});
const Order = mongoose.model('Order', OrderSchema);

// --- 8. OrderItem Schema ---
const OrderItemSchema = new Schema({
    product_id: { type: mongoose.Schema.Types.ObjectId, ref: 'Product', required: true },

    unit_price: { type: Number, required: true }

OrderSchema.virtual('items').add({ type: [OrderItemSchema] });

const cartService = {
    getCart: async (userId) => {
        const cart = await Cart.findOne({ user_id: userId }).populate('items.product_id');
        if (!cart) {
            // Initialize cart if it doesn't exist
            const newCart = new Cart({ user_id: userId });
            await newCart.save();
            return newCart;
        }
        return cart;
    },

    addItemToCart: async (userId, productId, quantity, unitPrice) => {
        const cart = await cartService.getCart(userId);
        const product = await Product.findById(productId);

        if (!product || product.stock_quantity < quantity) {
            throw new Error('Product out of stock or invalid product.');
        }

        // Check if item already exists in cart
        let existingItem = cart.items.find(item => item.product_id.toString() === productId);

        if (existingItem) {
            // Update quantity and ensure stock check is maintained
            if (existingItem.quantity + quantity > product.stock_quantity) {
                throw new Error('Cannot add this quantity. Stock limit exceeded.');
            }
            existingItem.quantity += quantity;
        } else {
            // Add new item
            cart.items.push({
                product_id: productId,
                quantity: quantity,
                unit_price: unitPrice
            });
        }

        await cart.save();
        return cart;
    },

    updateItemQuantity: async (userId, productId, newQuantity) => {
        const cart = await cartService.getCart(userId);
        const product = await Product.findById(productId);

        if (!product) {
            throw new Error('Product not found.');
        }

        if (newQuantity <= 0) {
            // If quantity is 0 or less, remove the item
            cart.items = cart.items.filter(item => item.product_id.toString() !== productId);
            await cart.save();
            return cart;
        }

        if (newQuantity > product.stock_quantity) {
            throw new Error('Requested quantity exceeds available stock.');
        }

        // Update the item quantity
        const item = cart.items.find(item => item.product_id.toString() === productId);
        if (item) {
            item.quantity = newQuantity;
            await cart.save();
        } else {
            throw new Error('Item not found in cart.');
        }
        return cart;
    },

    removeItemFromCart: async (userId, productId) => {
        const cart = await cartService.getCart(userId);
        const initialLength = cart.items.length;
        
        // Filter out the item to be removed
        cart.items = cart.items.filter(item => item.product_id.toString() !== productId);

        if (cart.items.length === initialLength) {
            throw new Error('Item not found in cart.');
        }

        await cart.save();
        return cart;
    }
};

const orderService = {
    checkout: async (userId, cartItems, shippingAddress) => {
        // 1. Validate Stock and Calculate Total
        let totalAmount = 0;
        const itemsToOrder = [];

        // Transaction start
        const session = await mongoose.startSession();
        session.startTransaction();

        try {
            // 2. Check Stock and Prepare Order Items
            for (const item of cartItems) {
                const product = await Product.findById(item.product_id).session(session);
                if (!product || product.stock_quantity < item.quantity) {
                    throw new Error(`Insufficient stock for ${product ? product.name : 'product'}.`);
                }
                
                // Calculate item cost
                const itemTotal = product.price * item.quantity;
                totalAmount += itemTotal;
                
                itemsToOrder.push({
                    product_id: product._id,
                    quantity: item.quantity,
                    unit_price: product.price
                });
            }

            // 3. Create Order
            const newOrder = new Order({
                user_id: userId,
                status: 'processing',
                total_amount: totalAmount,
                shipping_address: shippingAddress
            });
            await newOrder.save({ session });

            // 4. Update Stock (Crucial step)
            for (const item of itemsToOrder) {
                await Product.findByIdAndUpdate(
                    item.product_id,
                    { $inc: { stock_quantity: -item.quantity } },
                    { session }
                );
            }

            // 5. Create Order Items Records
            const orderItemRecords = itemsToOrder.map(item => ({
                product_id: item.product_id,
                quantity: item.quantity,
                unit_price: item.unit_price
            }));
            await OrderItemModel.insertMany(orderItemRecords, { session });

            // 6. Process Payment (Mocked)
            const paymentReference = `TXN-${Date.now()}`;
            const newPayment = new Payment({
                order_id: newOrder._id,
                method: 'mock_card',
                status: 'success',
                transaction_reference: paymentReference
            });
            await newPayment.save({ session });

            // 7. Commit Transaction
            await session.commitTransaction();
            await session.endSession();

            return {
                order: newOrder,
                payment: newPayment,
                message: 'Order placed successfully.'
            };

        } catch (error) {
            await session.abortTransaction();
            throw error;
        }
    },

    getOrderHistory: async (userId) => {
        const orders = await Order.find({ user_id: userId }).populate('items');
        return orders;
    }
};
