const { Product, Cart, CartItem, Order, OrderItem, Payment } = require('../models');
const mongoose = require('mongoose');

/**
 * Helper function to calculate total cost of items in the cart.
 * @param {Array} items - Array of CartItem documents.
 * @returns {number} Total amount.
 */
const calculateCartTotal = async (items) => {
    let total = 0;
    for (const item of items) {
        total += item.quantity * item.unit_price;
    }
    return parseFloat(total.toFixed(2));
};

/**
 * FR-001: Browse Products
 * @param {Object} filters - Filtering criteria (e.g., categoryId, minPrice).
 * @param {Object} sort - Sorting criteria (e.g., sortBy, direction).
 * @returns {Promise<Array>} List of products.
 */
const getProducts = async (filters = {}, sort = {}) => {
    let query = {};
    let filterConditions = [];

    if (filters.categoryId) {
        query.category_id = filters.categoryId;
        filterConditions.push({ category_id: filters.categoryId });
    }
    if (filters.minPrice) {
        filterConditions.push({ price: { $gte: parseFloat(filters.minPrice) } });
    }
    if (filters.maxPrice) {
        filterConditions.push({ price: { $lte: parseFloat(filters.maxPrice) } });
    }

    if (filterConditions.length > 0) {
        query = { $and: filterConditions };
    }

    const productList = await Product.find(query)
        .select('name description price image_url stock_quantity')
        .sort(sort.sortBy ? { [sort.sortBy]: sort.direction === 'asc' ? 1 : -1 } : { createdAt: -1 })
        .limit(sort.limit || 20);

    return productList;
};

/**
 * FR-002: View Product Details
 * @param {string} productId - ID of the product.
 * @returns {Promise<Object>} Detailed product information.
 */
const getProductDetails = async (productId) => {
    const product = await Product.findById(productId).select('-stock_quantity');
    if (!product) {
        throw new Error('Product not found');
    }
    return product;
};

/**
 * FR-003: Manage Shopping Cart - Get Cart
 * @param {string} userId - ID of the user.
 * @returns {Promise<{cart: Object, items: Array}>} Cart details and items.
 */
const getCartDetails = async (userId) => {
    const cart = await Cart.findOne({ user_id: userId });
    if (!cart) {
        throw new Error('Cart not found');
    }
    const items = await CartItem.find({ cart_id: cart._id }).populate('product_id', 'name price image_url');
    
    const cartItemsWithDetails = await Promise.all(items.map(async (item) => {
        const product = await Product.findById(item.product_id);
        return {
            ...item.toObject(),
            productName: product ? product.name : 'Unknown Product',
            productImage: product ? product.image_url : null,
            productPrice: product ? product.price : item.unit_price
        };
    }));

    return { cart: cart, items: cartItemsWithDetails };
};

/**
 * FR-003: Manage Shopping Cart - Add/Update Item
 * @param {string} userId - ID of the user.
 * @param {string} productId - ID of the product.
 * @param {number} quantity - Quantity to add/update.
 * @returns {Promise<CartItem>} Updated cart item.
 */
const updateCartItem = async (userId, productId, quantity) => {
    const cart = await Cart.findOne({ user_id: userId });
    if (!cart) {
        throw new Error('Cart not found');
    }

    const product = await Product.findById(productId);
    if (!product || product.stock_quantity < quantity) {
        throw new Error('Product unavailable or insufficient stock.');
    }

    let cartItem = await CartItem.findOne({ cart_id: cart._id, product_id: productId });

    if (cartItem) {
        // Update quantity
        cartItem.quantity += quantity;
        await cartItem.save();
    } else {
        // Create new item
        const unitPrice = product.price;
        const newItem = new CartItem({
            cart_id: cart._id,
            product_id: productId,
            quantity: quantity,
            unit_price: unitPrice
        });
        await newItem.save();
    }
    return await CartItem.findOne({ cart_id: cart._id, product_id: productId });
};

/**
 * FR-003: Manage Shopping Cart - Remove Item
 * @param {string} cartId - ID of the cart.
 * @param {string} productId - ID of the product to remove.
 * @returns {Promise<Boolean>} Success status.
 */
const removeItemFromCart = async (cartId, productId) => {
    const result = await CartItem.deleteOne({ cart_id: cartId, product_id: productId });
    if (result.deletedCount === 0) {
        throw new Error('Item not found in cart.');
    }
    // Optionally, clear the cart if it becomes empty
    // await Cart.findByIdAndUpdate(cartId, { $set: 'empty' });
    return true;
};

/**
 * FR-004 & FR-005: Checkout Process & Place Order
 * @param {string} userId - ID of the user.
 * @param {Object} shippingAddressData - Address details.
 * @param {string} paymentMethod - Payment method used.
 * @param {string} paymentToken - Mock payment token/reference.
 * @returns {Promise<{order: Object, payment: Object}>} Confirmed order and payment details.
 */
const placeOrder = async (userId, shippingAddressData, paymentMethod, paymentToken) => {
    // 1. Get current cart items
    const cartDetails = await getCartDetails(userId);
    const cartItems = cartDetails.items;

    if (cartItems.length === 0) {
        throw new Error('Cannot place order: Cart is empty.');
    }

    // 2. Calculate totals (BR-004)
    const totalAmount = await calculateCartTotal(cartItems);
    
    // 3. Create Order (BR-006: Snapshotting)
    const orderItems = cartItems.map(item => ({
        product_id: item.product_id,
        quantity: item.quantity,
        unit_price: item.productPrice
    }));

    const newOrder = new Order({
        user_id: userId,
        status: 'pending', // Initial status
        total_amount: totalAmount,
        shipping_address: shippingAddressData.id // Assuming address ID is passed
    });
    const savedOrder = await newOrder.save();

    // 4. Create OrderItems
    const orderItemPromises = orderItems.map(item => new OrderItem({
        order_id: savedOrder._id,
        product_id: item.product_id,
        quantity: item.quantity,
        unit_price: item.unit_price
    })).then(item => item.save());
    await Promise.all(orderItemPromises);

    // 5. Process Payment (Mocked)
    const payment = new Payment({
        order_id: savedOrder._id,
        method: paymentMethod,
        status: 'paid', // Assume success for MVP
        transaction_reference: paymentToken
    });
    const savedPayment = await payment.save();

    // 6. Update Inventory (BR-001)
    const inventoryUpdatePromises = cartItems.map(async (item) => {
        await Product.findByIdAndUpdate(
            item.product_id,
            { $inc: { stock_quantity: -item.quantity } }
        );
    });
    await Promise.all(inventoryUpdatePromises);

    // 7. Update Order Status (BR-003)
    await Order.findByIdAndUpdate(savedOrder._id, { status: 'processing' });

    // 8. Clear Cart (BR-002)
    await CartItem.deleteMany({ cart_id: cartDetails.cart._id });
    
    return { order: savedOrder, payment: savedPayment };
};

/**
 * FR-006: View Order History
 * @param {string} userId - ID of the user.
 * @returns {Promise<Array>} List of orders.
 */
const getOrderHistory = async (userId) => {
    const orders = await Order.find({ user_id: userId })
        .populate('shipping_address', 'line1 city postal_code country')
        .sort({ created_at: -1 });
    
    return orders;
};

/**
 * FR-007: Admin Product Management - Create Product
 * @param {Object} productData - Product details.
 * @returns {Promise<Product>} Created product.
 */
const createProduct = async (productData) => {
    // Basic validation and default setting
    const newProduct = new Product({
        category_id: productData.category_id,
        name: productData.name,
        description: productData.description,
        price: productData.price,
        stock_quantity: productData.stock_quantity || 0,
        image_url: productData.image_url,
        is_active: productData.is_active !== undefined ? productData.is_active : true
    });
    return newProduct.save();
};

/**
 * FR-007: Admin Product Management - Update Product
 * @param {string} productId - ID of the product.
 * @param {Object} updateData - Fields to update.
 * @returns {Promise<Product>} Updated product.
 */
const updateProduct = async (productId, updateData) => {
    const updatedProduct = await Product.findByIdAndUpdate(
        productId,
        { $set: updateData },
        { new: true, runValidators: true }
    );
    if (!updatedProduct) {
        throw new Error('Product not found');
    }
    return updatedProduct;
};

/**
 * FR-007: Admin Product Management - Delete Product
 * @param {string} productId - ID of the product.
 * @returns {Promise<Boolean>} Deletion status.
 */
const deleteProduct = async (productId) => {
    const result = await Product.deleteOne({ _id: productId });
    if (result.deletedCount === 0) {
        throw new Error('Product not found');
    }
    return true;
};
